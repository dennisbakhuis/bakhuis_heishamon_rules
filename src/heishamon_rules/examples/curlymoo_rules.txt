on System#Boot then
	#WPAan = 0;
	#WPUit = 0;
	#prevSilentMode = 0;
	#silentMode = 0;
	#maxShift = 0;
	#maxTa = 0;
	#onTemp = 0;
	#outletOnTemp = 0;
	#prevOutletTemp = 0;
	#varCounter1 = 0;
	#varCounter2 = 0;
	#totalRunTime = 0;
	#allowSilentMode = 1;
	#allowShift = 1;

	setTimer(3, 60);
end

on setThermostaat then
	#T1 = ?setpoint + 0.20;
	#T2 = ?setpoint - 0.20;
end

on timer=2 then
	#allowSilentMode = 1;
end

on checkSilentMode then
	if isset(@Outside_Temp) == 1 then
		if #allowSilentMode == 1 then
			if @Outside_Temp < 2 then
				if %hour > 22 || %hour < 7 then
					#silentMode = 1;
				else
					#silentMode = 0;
				end
			else
				if @Outside_Temp < 5 then
					#silentMode = 1;
				else
					if @Outside_Temp < 10 then
						#silentMode = 2;
					else
						#silentMode = 3;
					end
				end
			end
			if #prevSilentMode != #silentMode then
				setTimer(2, 900);
				#allowSilentMode = 0;
				#prevSilentMode = #silentMode;
				#varCounter2 = coalesce(#varCounter2, 0) + 1;
				@SetQuietMode = #silentMode;
			end
		end
	end
end

on stooklijn then
	$diff = 0;
	$offset = 0;
	if @Outside_Temp < 2 then
		$offset = -0.2;
	end
	if #targetTempDiff <= (0 + $offset) then
			if (#maxTa - @Main_Outlet_Temp) >= 3 then
				$diff = -4;
			else
				$diff = -3;
			end
	else
		if #targetTempDiff <= (0.2 + $offset) then
			if (#maxTa - @Main_Outlet_Temp) >= 1 then
				$diff = -2;
			else
				$diff = -1;
			end
		else
			if #targetTempDiff >= (0.4 + $offset) then
				$diff = 1;
			end
		end
	end
	if $diff != 0 && isset(@Z1_Heat_Curve_Target_Low_Temp) && isset(@Z1_Heat_Curve_Target_High_Temp) then
		if (#maxTa + $diff) >= @Z1_Heat_Curve_Target_Low_Temp && (#maxTa + $diff) <= @Z1_Heat_Curve_Target_High_Temp then
			#maxShift = $diff;
		else
			if $diff < 0 then
				if (#maxTa + $diff) >= @Z1_Heat_Curve_Target_Low_Temp then
					#maxShift = max(min((#maxTa - @Main_Outlet_Temp), 0), -1);
				else
					#maxShift = min((@Z1_Heat_Curve_Target_Low_Temp - #maxTa), (#maxTa - @Main_Target_Temp), 0);
				end
			else
				if $diff > 0 && (#maxTa + $diff) > @Z1_Heat_Curve_Target_High_Temp then
					#maxShift = 0;
				end
			end
		end
	end
end

on berekenWar then
	if isset(@Outside_Temp) && isset(@Z1_Heat_Curve_Outside_High_Temp) && isset(@Z1_Heat_Curve_Target_Low_Temp) && isset(@Z1_Heat_Curve_Outside_Low_Temp) && isset(@Z1_Heat_Curve_Target_High_Temp) then
		if @Outside_Temp >= @Z1_Heat_Curve_Outside_High_Temp then
			#maxTa = @Z1_Heat_Curve_Target_Low_Temp;
		else
			if @Outside_Temp <= @Z1_Heat_Curve_Outside_Low_Temp then
				#maxTa = @Z1_Heat_Curve_Target_High_Temp;
			else
				#maxTa = ceil(@Z1_Heat_Curve_Target_Low_Temp + ((@Z1_Heat_Curve_Outside_High_Temp - @Outside_Temp) * (@Z1_Heat_Curve_Target_High_Temp - @Z1_Heat_Curve_Target_Low_Temp) / (@Z1_Heat_Curve_Outside_High_Temp - @Z1_Heat_Curve_Outside_Low_Temp)));
			end
		end
	end

	stooklijn();
end

on timer=6 then
	#allowShift = 1;
end

on timer=5 then
	#maxShift = max(min((#maxTa - @Main_Outlet_Temp), 0), -1);
	setTimer(6, 900);
	setNewTemp();
end

on setNewTemp then
	if isset(@Main_Outlet_Temp) && isset(@Main_Inlet_Temp) && isset(@Main_Target_Temp) && isset(?setpoint) && isset(@High_Pressure) then
		#dT = @Main_Outlet_Temp - @Main_Inlet_Temp;
		if #maxShift < 0 && ((#dT < 3 && #outletOnTemp == 0) || @High_Pressure < 15) then
			#maxShift = 0;
		end
		#dTOutlet = @Main_Outlet_Temp - (#maxTa + #maxShift);
		if #dTOutlet > 1 && @Defrosting_State == 0 && #allowShift == 1 then
			setTimer(5, 90);
			#allowShift = 0;
		end
		if ?state == 0 then
			#maxShift = 0;
		end
		if #prevShift != #maxShift || @Z1_Heat_Request_Temp != #maxShift then
			#prevShift = #maxShift;
			#varCounter1 = coalesce(#varCounter1, 0) + 1;
			@SetZ1HeatRequestTemperature = #maxShift;
		end
	end
end

on resetTeller then
	#tellerAan = 0;
	#tellerUit = 0;
end

on resetDummy then
	#WPAan = 0;
	#WPUit = 0;
	resetTeller();
end

on schakelWP then
	if #WPUit == 1 then
		WPUit();
	end
	if #WPAan == 1 then
		WPAan();
	end

	resetTeller();
	resetDummy();
end

on ?state then
	if ?state == 0 then
		WPUit();
	end
	if ?state == 1 then
		WPAan();
	end
end

on WPUit then
	#runTime = 0;
	#WPAan = 0;
	#WPUit = 2;
	#Teller = 0;
	#outletOnTemp = 0;
end

on WPAan then
	#WPAan = 2;
	#WPUit = 0;
end

on @Main_Outlet_Temp then
	if @Main_Outlet_Temp >= #maxTa	then
		#outletOnTemp = 1;
	end
	#prevOutletTemp = @Main_Outlet_Temp;
	berekenWar();
	setNewTemp();
end

on @Main_Inlet_Temp then
	berekenWar();
	setNewTemp();
end


on ?Temperature then
	setThermostaat();
end

on timer=3 then
	#Temp = ?Temperature;
	#setpoint = ?setpoint;

	#targetTempDiff = #setpoint - #Temp;

	if #Temp > #T1 then
		#onTemp = 1;
	else
		if #Temp < #T2 then
			#onTemp = 0;
		else
			setOnTemp();
			resetDummy();
		end
	end

	if @Outside_Temp > 0 then
		if @Defrosting_State == 0 then
			if @Compressor_Freq == 0 && ?state == 1 && @Outside_Temp >= 6 then
				if #onTemp == 3 then
					#WPAan = 0;
					#WPUit = 1;
					schakelWP();
				else
					#outletOnTemp = 0;
				end
			end
		end
	end

	berekenWar();
	setNewTemp();
	checkSilentMode();

	if ?state == 1 then
		#runTime = coalesce(#runTime, 0) + 1;
	end

	#totalRunTime = coalesce(#totalRunTime, 0) + 1;

	setTimer(3, 60);
end
